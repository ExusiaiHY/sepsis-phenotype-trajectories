\subsection{Detailed Competitor Analysis}

\begin{competitorbox}{Multimodal-ICU-Mortality [1]}
\textbf{Source:} GitHub (bbleier) \hfill \textbf{AUROC:} 0.82

\textbf{Architecture:}
\begin{itemize}[nosep,leftmargin=1em]
    \item \textbf{Notes:} ClinicalBERT fine-tuned for classification
    \item \textbf{CXR:} DenseNet121 (CheXnet pretrained)
    \item \textbf{Labs:} Fully connected network
    \item \textbf{Fusion:} Concatenation with adaptors
\end{itemize}

\textbf{Performance:} Notes-only: 0.75, Labs-only: 0.74, CXR-only: 0.67

\textbf{Limitations vs. Ours:}
\begin{itemize}[nosep,leftmargin=1em]
    \item Requires imaging (delayed/unavailable in ICU)
    \item Requires clinical notes (NLP preprocessing)
    \item Static classification, no temporal trajectory
\end{itemize}

\textbf{Our Advantage:} 0.873 AUROC with structured EHR only, no imaging/text dependency
\end{competitorbox}

\vspace{0.3cm}

\begin{competitorbox}{MedFuse [2]}
\textbf{Source:} NYU Abu Dhabi - CAI \hfill \textbf{AUROC:} 0.81-0.84

\textbf{Architecture:} Two-stage: modality-specific pretraining $\rightarrow$ LSTM-based fusion

\textbf{Fusion Strategies:} Early, Joint, DAFT, MMTM, MedFuse (proposed)

\textbf{Key Innovation:} Robustness on partially paired test sets with missing CXR

\textbf{Limitations vs. Ours:}
\begin{itemize}[nosep,leftmargin=1em]
    \item LSTM less expressive than Transformer for long sequences
    \item Static output (single risk score)
    \item No explicit phenotype modeling
    \item Limited external validation
\end{itemize}

\textbf{Our Advantage:} Masked Transformer + rolling-window phenotypes; validated on 295K patients
\end{competitorbox}

\vspace{0.3cm}

\begin{competitorbox}{CRL-MMNAR [3]}
\textbf{Source:} EMNLP 2025 \hfill \textbf{AUROC:} $\sim$0.85

\textbf{Architecture (Two-Stage):}
\begin{enumerate}[nosep,leftmargin=1em]
    \item \textbf{Representation:} Missingness-aware transformation + attention fusion
    \item \textbf{Prediction:} Multitask heads + rectifier for bias correction
\end{enumerate}

\textbf{Key Innovations:} Missingness embeddings, causal debiasing, cross-modality reconstruction

\textbf{Limitations vs. Ours:}
\begin{itemize}[nosep,leftmargin=1em]
    \item Complex two-stage training
    \item No temporal trajectory analysis
    \item No phenotype stratification
    \item No treatment effect estimation
\end{itemize}

\textbf{Our Advantage:} Simpler training, lower missingness sensitivity ($|r|$=0.148), temporal trajectories
\end{competitorbox}

\vspace{0.3cm}

\begin{competitorbox}{MIMIC-Multimodal (DiReCT) [4]}
\textbf{Source:} Duke-NUS \hfill \textbf{AUROC:} 0.83-0.87

\textbf{Architecture:} Benchmark of 12 foundation models
\begin{itemize}[nosep,leftmargin=1em]
    \item Time-series: GRU, moment-based
    \item Images: CXR-Foundation, Swin Transformer
    \item Notes: OpenAI embeddings, RadBERT
    \item VLM: GPT-4o-mini, LLaVA, Gemini, MedGemma
\end{itemize}

\textbf{Limitations vs. Ours:}
\begin{itemize}[nosep,leftmargin=1em]
    \item Computationally expensive (12 models)
    \item Black-box LLM/VLM not interpretable
    \item No temporal modeling
    \item Static risk scores only
\end{itemize}

\textbf{Our Advantage:} Lightweight single encoder, interpretable trajectories, comparable AUROC (0.873)
\end{competitorbox}
